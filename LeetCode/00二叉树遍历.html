# 二叉树是由根节点，左子树，右子树组成；左子树和右子树分别也是二叉树。

//一个二叉树例子

  var tree = {
    value: 1,
    left: {
      value: 2,
      left: {
        value: 4
      }
    },
    right: {
      value: 3,
      left: {
        value: 5,
        left: {
          value: 7
        },
        right: {
          value: 8
        }
      },
      right: {
        value: 6
      }
    }
  };

  //遍历二叉树有广度优先遍历，和深度优先遍历；
  /*
  * 广度优先遍历(BFS)：
  * 1.新建一个队列，把根节点挨个入队；
  * 2.把队头出队并访问节点值；
  * 3.把队头的左右节点children挨个入队；
  * 4.重复2，3直到队列为空。
  * */
  //Polyfill
  let bfs = (tree) =>  {
    if(!tree) throw new Error('Empty Tree');
    const q = [tree];
    while(q.length) {
      const n = q.shift();
      console.log(n.value);
      if(n.left) q.push(n.left);
      if(n.right) q.push(n.right);
    }
  };
  //bfs(tree);

  /*
  * 深度优先遍历(DFS)：
  * 1.访问根节点；
  * 2.对根节点的children挨个遍历
  * D:访问根节点，L:遍历根节点的左子树，R:遍历根节点的右子树。
  * 先序遍历：DLR
  * 中序遍历：LDR
  * 后序遍历：LRD
  * */
  let dlr = (tree) => { //DLR
    if(tree) {
      console.log(tree.value);
      dlr(tree.left);
      dlr(tree.right);
    }
  }
  //dlr(tree); //1,2,4,3,5,7,8,6

let ldr = (tree) => { //LDR
  if(tree) {
    ldr(tree.left);
    console.log(tree.value);
    ldr(tree.right);
  }
}
//ldr(tree); //4,2,1,7,5,8,3,6

let lrd = (tree) => { //LRD
  if(tree) {
    lrd(tree.left);
    lrd(tree.right);
    console.log(tree.value);
  }
}
//lrd(tree); //4,2,7,8,5,6,3,1
//同样还有不适用递归的方式实现深度优先遍历

//DLR
let dlr2 = (tree) => {
  if(!tree) throw new Error('Empty tree');
  const stack = [tree];
  while(stack.length) {
    const n = stack.pop();
    console.log(n.value);
    if(n.right) stack.push(n.right);
    if(n.left) stack.push(n.left); //left放在后面，先处理left子树
  }
};
//dlr2(tree);

//LDR
let ldr2 = (tree) => {
  if(!tree) throw new Error('Empty tree');
  const stack = [];
  while(stack.length || tree) {
    if(tree){
      stack.push(tree);
      tree = tree.left;
    }else {
      tree = stack.pop();
      console.log(tree.value);
      tree = tree.right;
    }

  }
}
//ldr2(tree);

//LRD
let lrd2 = (tree) => {
  if(tree) {
    const stack1 = [tree];
    const stack2 = [];
    while(stack1.length) {
      const n = stack1.pop();
      stack2.push(n);
      if(n.left) stack1.push(n.left);
      if(n.right) stack1.push(n.right);
    }
    while(stack2.length) {
      console.log(stack2.pop().value);
    }
  }
}
